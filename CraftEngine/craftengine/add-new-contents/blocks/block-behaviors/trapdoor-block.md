```
blocks:
  default:palm_trapdoor:
    behavior:
      type: trapdoor_block
      can-open-with-hand: true
      can-open-by-wind-charge: true
      sounds:
        open: block.wooden_trapdoor.open
        close: block.wooden_trapdoor.close
```
